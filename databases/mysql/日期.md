- 当前时间
	SELECT NOW() date FROM DUAL

- 当前日期
	SELECT CURDATE() date FROM DUAL		-- 00:00:00

- 之前日期
	SELECT DATE_SUB(CURDATE(),INTERVAL 1 DAY) date FROM DUAL		--YEAR, MONTH, DAY, HOUR, MINUTE, SECOND

- 之后日期
	SELECT DATE_SUB(CURDATE(),INTERVAL -1 DAY) date FROM DUAL		--YEAR, MONTH, DAY, HOUR, MINUTE, SECOND
